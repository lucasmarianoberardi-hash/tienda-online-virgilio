        let allProducts = [];
        const DISCOUNT_THRESHOLD = 10;
        const DISCOUNT_PERCENTAGE = 0.10;

        async function loadProducts() {
            try {
                const response = await fetch(productsUrl);
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const csvData = await response.text();
                allProducts = parseCSV(csvData);

                displayProducts(allProducts);
                generateCategoryMenu(allProducts);
                setupSmoothScrolling();
            } catch (error) {
                console.error('🔴 Error al cargar los productos:', error);
                document.getElementById('products').innerHTML = '<p class="error">Error al cargar los productos. Por favor, intenta nuevamente más tarde.</p>';
            }
        }

        function parseCSV(csvText) {
            const lines = csvText.trim().split('\n');
            if (lines.length <= 1) {
                return [];
            }
            const headers = lines[0].split(',').map(header => header.trim());
            const products = [];
            for (let i = 1; i < lines.length; i++) {
                const values = lines[i].split(',');
                const product = {};
                if (values.length !== headers.length) {
                    continue;
                }
                for (let j = 0; j < headers.length; j++) {
                    product[headers[j]] = values[j] ? values[j].trim() : '';
                }
                products.push(product);
            }
            return products;
        }

        function generateCategoryMenu(products) {
            const categoryMenu = document.querySelector('.menu-desplegable-contenido ul');
            const uniqueCategories = [...new Set(products.map(product => product.Categoría).filter(cat => cat && cat.trim() !== ''))];
            categoryMenu.innerHTML = '';

            uniqueCategories.sort().forEach(category => {
                const listItem = document.createElement('li');
                const link = document.createElement('a');
                link.href = `#${normalizeText(category)}`;
                link.textContent = category.replace(/🌻|🫚|🫘|❄️|-/g, '').trim();
                listItem.appendChild(link);
                categoryMenu.appendChild(listItem);
            });
        }

        function displayProducts(productsToDisplay) {
            const productsContainer = document.getElementById('products');
            productsContainer.innerHTML = '';

            const categories = {};
            productsToDisplay.forEach(product => {
                if (product.Categoría && product.Categoría.trim() !== '') {
                    if (!categories[product.Categoría]) {
                        categories[product.Categoría] = [];
                    }
                    categories[product.Categoría].push(product);
                }
            });

            Object.keys(categories).sort().forEach(categoryId => {
                const normalizedCategoryId = normalizeText(categoryId);
                let categoryDiv = document.createElement('div');
                categoryDiv.classList.add('category');
                categoryDiv.id = normalizedCategoryId;

                const h2 = document.createElement('h2');
                h2.textContent = categoryId;
                categoryDiv.appendChild(h2);

                const categoryProductsDiv = document.createElement('div');
                categoryProductsDiv.classList.add('category-products');
                categoryDiv.appendChild(categoryProductsDiv);

                productsContainer.appendChild(categoryDiv);

                categories[categoryId].forEach(productData => {
                    const productoDiv = document.createElement('div');
                    productoDiv.classList.add('producto');
                    productoDiv.dataset.nombre = productData['Nombre del Producto'];

                    const img = document.createElement('img');
                    const imageUrl = productData['URL de la Imagen'];
                    img.src = imageUrl;
                    img.alt = productData['Nombre del Producto'] || 'Producto';
                    productoDiv.appendChild(img);

                    img.onerror = function() {
                        this.src = 'https://via.placeholder.com/150?text=Imagen+No+Disp.';
                        this.alt = 'Imagen no disponible';
                        this.style.border = '2px dashed red';
                    };

                    const productoInfoDiv = document.createElement('div');
                    productoInfoDiv.classList.add('producto-info');

                    const h3 = document.createElement('h3');
                    h3.textContent = productData['Nombre del Producto'];
                    productoInfoDiv.appendChild(h3);

                    const precioP = document.createElement('p');
                    const price = parseFloat(productData.Precio);
                    const isInStock = price > 0;
                    precioP.textContent = isInStock ? `$${price.toFixed(2)}` : 'Sin Stock';
                    if (!isInStock) {
                        precioP.classList.add('no-stock');
                    }
                    productoInfoDiv.appendChild(precioP);

                    const propiedadesContainer = document.createElement('div');
                    propiedadesContainer.classList.add('propiedades-container');
                    if (productData.SinTACC === 'Sí') {
                        propiedadesContainer.innerHTML += `<img src="${logoSinTaccUrl}" alt="Logo Sin TACC" class="logo-propiedad">`;
                    }
                    if (productData.Vegano === 'Sí') {
                        propiedadesContainer.innerHTML += `<img src="${logoVeganoUrl}" alt="Logo Vegano" class="logo-propiedad">`;
                    }
                    if (productData.SinAzucar === 'Sí') {
                        propiedadesContainer.innerHTML += `<img src="${logoSinAzucarUrl}" alt="Logo Sin Azúcar" class="logo-propiedad">`;
                    }
                    if (productData.SinSal === 'Sí') {
                        propiedadesContainer.innerHTML += `<img src="${logoSinSalUrl}" alt="Logo Sin Sal" class="logo-propiedad">`;
                    }
                    productoInfoDiv.appendChild(propiedadesContainer);

                    const descripcionContainer = document.createElement('div');
                    descripcionContainer.classList.add('descripcion-container');
                    const descripcionCorta = document.createElement('span');
                    descripcionCorta.classList.add('descripcion-corta');
                    descripcionCorta.textContent = productData.Descripción.length > 50 ? productData.Descripción.substring(0, 50) + '...' : productData.Descripción;
                    descripcionContainer.appendChild(descripcionCorta);

                    if (productData.Descripción.length > 50) {
                        const leerMasBtn = document.createElement('button');
                        leerMasBtn.classList.add('leer-mas-btn');
                        leerMasBtn.innerHTML = '<i class="fas fa-plus"></i>';
                        leerMasBtn.onclick = (e) => {
                            e.stopPropagation();
                            const fullDesc = descripcionContainer.querySelector('.descripcion-completa');
                            const shortDesc = descripcionContainer.querySelector('.descripcion-corta');
                            if (fullDesc) {
                                fullDesc.remove();
                                shortDesc.style.display = 'block';
                                leerMasBtn.innerHTML = '<i class="fas fa-plus"></i>';
                            } else {
                                const newFullDesc = document.createElement('span');
                                newFullDesc.classList.add('descripcion-completa');
                                newFullDesc.textContent = productData.Descripción;
                                descripcionContainer.appendChild(newFullDesc);
                                shortDesc.style.display = 'none';
                                leerMasBtn.innerHTML = '<i class="fas fa-minus"></i>';
                            }
                        };
                        descripcionContainer.appendChild(leerMasBtn);
                    }
                    productoInfoDiv.appendChild(descripcionContainer);

                    productoDiv.appendChild(productoInfoDiv);

                    const actionsDiv = document.createElement('div');
                    actionsDiv.classList.add('producto-actions');

                    if (isInStock) {
                        const cantidadInput = document.createElement('input');
                        cantidadInput.type = 'number';
                        cantidadInput.value = 1;
                        cantidadInput.min = 1;
                        actionsDiv.appendChild(cantidadInput);

                        const addButton = document.createElement('button');
                        addButton.textContent = 'Añadir';
                        addButton.onclick = () => addToCart(productData, parseInt(cantidadInput.value));
                        actionsDiv.appendChild(addButton);
                    } else {
                        const noStockButton = document.createElement('button');
                        noStockButton.textContent = 'Sin Stock';
                        noStockButton.classList.add('no-stock-button');
                        noStockButton.disabled = true;
                        actionsDiv.appendChild(noStockButton);
                    }

                    productoDiv.appendChild(actionsDiv);
                    categoryProductsDiv.appendChild(productoDiv);
                });
            });
        }

        function showNotification(message) {
            const notification = document.createElement('div');
            notification.classList.add('notification');
            notification.textContent = message;
            document.body.appendChild(notification);

            setTimeout(() => {
                notification.remove();
            }, 3000);
        }

        function normalizeText(text) {
            return text.normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/\s+/g, '-').toLowerCase();
        }

        function addToCart(product, quantity) {
            if (quantity < 1 || isNaN(quantity)) {
                showNotification('Cantidad no válida.');
                return;
            }

            const existingItem = cart.find(item => item.product['Nombre del Producto'] === product['Nombre del Producto']);

            if (existingItem) {
                existingItem.quantity += quantity;
            } else {
                cart.push({
                    product: product,
                    quantity: quantity
                });
            }
            showNotification(`${quantity}x ${product['Nombre del Producto']} añadido al carrito.`);
            saveCart();
            updateCartDisplay();
        }

        function removeFromCart(productName) {
            cart = cart.filter(item => item.product['Nombre del Producto'] !== productName);
            saveCart();
            updateCartDisplay();
        }

        function updateCartQuantity(productName, newQuantity) {
            const item = cart.find(i => i.product['Nombre del Producto'] === productName);
            if (item) {
                item.quantity = Math.max(1, parseInt(newQuantity));
                saveCart();
                updateCartDisplay();
            }
        }

        function saveCart() {
            localStorage.setItem('cart', JSON.stringify(cart));
        }

        function updateCartDisplay() {
            const cartItemsList = document.getElementById('cart-items');
            cartItemsList.innerHTML = '';
            let subtotal = 0;
            let totalItems = 0;

            cart.forEach(item => {
                const li = document.createElement('li');
                li.classList.add('cart-item');
                const price = parseFloat(item.product.Precio);
                const itemSubtotal = price * item.quantity;
                subtotal += itemSubtotal;
                totalItems += item.quantity;

                li.innerHTML = `
                <div class="cart-item-details">
                    <p>${item.product['Nombre del Producto']}</p>
                    <p>$${price.toFixed(2)} c/u</p>
                </div>
                <div class="cart-item-quantity">
                    <button class="quantity-button" data-action="decrease" data-product="${item.product['Nombre del Producto']}">-</button>
                    <input type="number" value="${item.quantity}" min="1" data-product="${item.product['Nombre del Producto']}">
                    <button class="quantity-button" data-action="increase" data-product="${item.product['Nombre del Producto']}">+</button>
                </div>
                <p>$${itemSubtotal.toFixed(2)}</p>
                <button class="remove-item-button" data-product="${item.product['Nombre del Producto']}">&times;</button>
            `;
                cartItemsList.appendChild(li);
            });

            document.getElementById('cart-count').textContent = totalItems;
            document.getElementById('cart-count').style.display = totalItems > 0 ? 'flex' : 'none';
            document.getElementById('cart-subtotal').textContent = subtotal.toFixed(2);

            const discountMessage = document.getElementById('cart-discount-message');
            let finalTotal = subtotal;

            if (totalItems >= DISCOUNT_THRESHOLD) {
                const discountAmount = subtotal * DISCOUNT_PERCENTAGE;
                finalTotal = subtotal - discountAmount;
                discountMessage.textContent = `¡Bonificación por volumen! Ahorras $${discountAmount.toFixed(2)}`;
                discountMessage.style.display = 'block';
            } else {
                discountMessage.style.display = 'none';
            }

            document.getElementById('cart-total').textContent = finalTotal.toFixed(2);
            document.getElementById('send-order-button').disabled = totalItems === 0;

            document.querySelectorAll('.remove-item-button').forEach(button => {
                button.addEventListener('click', (e) => {
                    removeFromCart(e.target.dataset.product);
                });
            });

            document.querySelectorAll('.cart-item-quantity input').forEach(input => {
                input.addEventListener('change', (e) => {
                    updateCartQuantity(e.target.dataset.product, e.target.value);
                });
            });

            document.querySelectorAll('.cart-item-quantity .quantity-button').forEach(button => {
                button.addEventListener('click', (e) => {
                    const productName = e.target.dataset.product;
                    const input = document.querySelector(`.cart-item-quantity input[data-product="${productName}"]`);
                    let newQuantity = parseInt(input.value);
                    if (e.target.dataset.action === 'increase') {
                        newQuantity++;
                    } else if (e.target.dataset.action === 'decrease') {
                        newQuantity--;
                    }
                    if (newQuantity >= 1) {
                        input.value = newQuantity;
                        updateCartQuantity(productName, newQuantity);
                    } else {
                        removeFromCart(productName);
                    }
                    if (newQuantity <= 0) {
                        removeFromCart(productName);
                    }
                });
            });
        }

            if (totalItems >= DISCOUNT_THRESHOLD) {
                const discountAmount = subtotal * DISCOUNT_PERCENTAGE;
                finalTotal = subtotal - discountAmount;
                discountMessage = `\n(Bonificación por volumen: Ahorras $${discountAmount.toFixed(2)})`;
            }

            window.open(whatsappUrl, '_blank');

            document.getElementById('cart').style.display = 'none';
            cart = [];
            saveCart();
            updateCartDisplay();
        });

        function setupSmoothScrolling() {
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const targetId = this.getAttribute('href').substring(1);
                    const targetElement = document.getElementById(targetId);

                    if (targetElement) {
                        window.scrollTo({
                            top: targetElement.offsetTop - 80,
                            behavior: 'smooth'
                        });
                    }
                });
            });
        }

        const qrScanIcon = document.getElementById('qr-scan-icon');
        const qrScannerContainer = document.getElementById('qr-scanner-container');
        let html5QrcodeScanner = null;

        async function startScanner() {
            // Oculta otras ventanas de la interfaz
            document.getElementById('cart').style.display = 'none';
            document.getElementById('search').style.display = 'none';
            document.querySelector('.menu-desplegable-contenido').style.display = 'none';

            html5QrcodeScanner = new Html5QrcodeScanner(
                "reader", {
                    fps: 10,
                    qrbox: {
                        width: 250,
                        height: 250
                    },
                    supportedScanFormats: [Html5QrcodeSupportedFormats.QR_CODE],
                    videoConstraints: {
                        facingMode: "environment" // Fuerza la cámara trasera
                    }
                },
                false
            );

            html5QrcodeScanner.render(onScanSuccess);
            qrScannerContainer.style.display = 'flex';
        }

        async function onScanSuccess(decodedText) {
            if (!html5QrcodeScanner) return;

            // Apaga el escáner inmediatamente
            await html5QrcodeScanner.stop();
            html5QrcodeScanner.clear();
            html5QrcodeScanner = null;

            // Oculta el contenedor del escáner
            qrScannerContainer.style.display = 'none';

            // Busca y añade el producto al carrito
            const matchingProduct = allProducts.find(p => p.QR === decodedText);
            if (matchingProduct) {
                addToCart(matchingProduct, 1);
            } else {
                showNotification('❌ Producto no encontrado.');
            }
        }

        async function stopScanner() {
            if (html5QrcodeScanner) {
                try {
                    await html5QrcodeScanner.stop();
                    html5QrcodeScanner.clear();
                    html5QrcodeScanner = null;
                } catch (err) {
                    console.error("Error al detener el escáner:", err);
                }
            }
            qrScannerContainer.style.display = 'none';
        }

        qrScanIcon.addEventListener('click', () => {
            if (html5QrcodeScanner) {
                stopScanner();
            } else {
                startScanner();
            }
        });
